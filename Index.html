<script>

    // --- CONFIGURATION: YOU WILL REPLACE THIS URL AFTER STEP 5 ---

    const GOOGLE_SHEETS_CSV_URL = 'https://docs.google.com/spreadsheets/d/e/YOUR_SHEET_ID/pub?output=csv';

    // -----------------------------------------------------------


    async function loadData() {

        const contentDiv = document.getElementById('content');

        const batchId = getBatchIdFromURL();


        if (!batchId) {

            contentDiv.innerHTML = '<div class="error">Invalid QR code. Please check the package.</div>';

            return;

        }


        try {

            const response = await fetch(GOOGLE_SHEETS_CSV_URL);

            if (!response.ok) throw new Error('Network response was not ok');

            const csvText = await response.text();

            const data = parseCSV(csvText);

            const product = findProductByBatchId(data, batchId);


            if (product) {

                displayProductInfo(product);

            } else {

                contentDiv.innerHTML = '<div class="error">Product not found. Please verify the Batch ID or contact support.</div>';

            }

        } catch (error) {

            console.error('Error loading data:', error);

            contentDiv.innerHTML = '<div class="error">Unable to load product information. Please check your internet connection.</div>';

        }

    }


    function getBatchIdFromURL() {

        const pathParts = window.location.pathname.split('/');

        let lastPart = pathParts[pathParts.length - 1];

        if (lastPart === '' || lastPart === 'farm-trace') {

            return null;

        }

        if (lastPart.endsWith('.html')) {

            lastPart = lastPart.replace('.html', '');

        }

        return decodeURIComponent(lastPart);

    }


    function parseCSV(csvText) {

        const lines = csvText.trim().split(/\r?\n/);

        if (lines.length < 2) return [];


        const headers = lines[0].split(',').map(h => h.trim().replace(/^"|"$/g, ''));

        const data = [];


        for (let i = 1; i < lines.length; i++) {

            const values = [];

            let insideQuotes = false;

            let currentValue = '';

            const line = lines[i];


            for (let j = 0; j < line.length; j++) {

                const char = line[j];

                if (char === '"') {

                    insideQuotes = !insideQuotes;

                } else if (char === ',' && !insideQuotes) {

                    values.push(currentValue.trim().replace(/^"|"$/g, ''));

                    currentValue = '';

                } else {

                    currentValue += char;

                }

            }

            values.push(currentValue.trim().replace(/^"|"$/g, ''));


            const row = {};

            headers.forEach((header, idx) => {

                row[header] = values[idx] || '';

            });

            data.push(row);

        }

        return data;

    }


    function findProductByBatchId(data, batchId) {

        return data.find(row => row.BatchID && row.BatchID.trim().toUpperCase() === batchId.trim().toUpperCase());

    }


    function formatDate(dateStr) {

        if (!dateStr) return 'Not specified';

        try {

            const date = new Date(dateStr);

            if (isNaN(date.getTime())) return dateStr;

            return date.toLocaleDateString('en-GB');

        } catch(e) {

            return dateStr;

        }

    }


    function displayProductInfo(product) {

        const getValue = (key, defaultValue = 'Not specified') => product[key] && product[key].trim() ? product[key] : defaultValue;


        const productType = getValue('ProductType');

        let typeIcon = '🌾';

        if (productType.toLowerCase().includes('chicken')) typeIcon = '🐔';

        else if (productType.toLowerCase().includes('egg')) typeIcon = '🥚';

        else if (productType.toLowerCase().includes('pepper')) typeIcon = '🌶️';


        const html = `

            <div class="info-card">

                <div class="info-row">

                    <div class="info-label">${typeIcon} Batch ID</div>

                    <div class="info-value"><strong>${getValue('BatchID')}</strong></div>

                </div>

                <div class="info-row">

                    <div class="info-label">📦 Product</div>

                    <div class="info-value">${productType}</div>

                </div>

                <div class="info-row">

                    <div class="info-label">🌱 Planted / Hatched</div>

                    <div class="info-value">${formatDate(getValue('ProductionDate'))}</div>

                </div>

                <div class="info-row">

                    <div class="info-label">✂️ Harvest / Slaughter</div>

                    <div class="info-value">${formatDate(getValue('HarvestDate'))}</div>

                </div>

                <div class="info-row">

                    <div class="info-label">🍽️ Feed Used</div>

                    <div class="info-value">${getValue('FeedUsed')}</div>

                </div>

                <div class="info-row">

                    <div class="info-label">⏳ Expiry Date</div>

                    <div class="info-value">${formatDate(getValue('ExpiryDate'))}</div>

                </div>

            </div>

        `;

        document.getElementById('content').innerHTML = html;

    }


    loadData();
